function CalDistance()

for index = 1:4
    si=index;
    sj=RelevantIndex(index);
    if sj~=-1 && urstopped(si)==1
        urstop(si)=0;
        if abs(position(sj)-position(si))>300 || dir(si)~=dir(sj)
            urstart(si)=1;
            urstopped(si)=0;
        end
    end
end

function stopDecision(a)

dis=int32(10000);
needstop=0;

for index=0:N-1
    if a~= index && dir(a+1)==dir(index+1)
            tdis=dir(a+1)*(position(index+1)-position(a+1));
            if tdis<=300 && tdis>0
                if tdis<dis
                        dis=tdis
                        RelevantIndex(a+1)=index+1;
                        needstop=1;                    
                end
            end           
    end
end

if needstop==1
    urstop(a+1) = 1;
end
